import requests
from datetime import datetime, timedelta

def get_aws_minute_data():
    # 1. API URL 및 인증키 설정
    url = "https://apihub.kma.go.kr/api/typ01/cgi-bin/url/nph-aws2_min"
    auth_key = "YOUR_API_KEY"  # 본인의 발급 키 입력
    station_id = "108"         # 서울 지점 (전체 지점은 "0")

    # 2. 🔥 현재 시간에서 3분을 빼서 안전하게 직전 데이터를 요청하도록 설정
    now = datetime.now()
    request_time = now - timedelta(minutes=3)
    
    # 기상청 API 포맷인 YYYYMMDDHHMM 문자열로 변환 (예: 202607070841)
    target_time = request_time.strftime("%Y%m%d%H%M")

    # 3. 파라미터 구성
    params = {
        'tm2': target_time,
        'stn': station_id,
        'disp': '1',
        'help': '0',
        'authKey': auth_key
    }

    print(f"⏰ [요청 시각] 현재 시간: {now.strftime('%Y-%m-%d %H:%M:%S')}")
    print(f"📡 [API 요청] 조회 대상 시각 (3분 전): {target_time}\n")

    try:
        response = requests.get(url, params=params)
        
        if response.status_code == 200:
            raw_text = response.text
            lines = raw_text.strip().split('\n')
            
            for line in lines:
                if line.startswith('#'): 
                    print(line)  # 헤더 출력
                else:
                    data_fields = line.split(',')
                    print(data_fields)  # 실시간 데이터 출력
        else:
            print(f"❌ 호출 실패 (Status Code): {response.status_code}")

    except Exception as e:
        print(f"❌ 에러 발생: {e}")

if __name__ == "__main__":
    get_aws_minute_data()
